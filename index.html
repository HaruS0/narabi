async function updateStatus(myNumber) { const res = await fetch('あなたのGASデプロイURL'); const data = await res.json(); // 待機組数を表示 document.getElementById('total').innerText = data.totalWaiting; // 呼び出し中の番号を判定 const isMyTurn = data.callingNumbers.includes(parseInt(myNumber)); const statusDiv = document.getElementById('calling'); if (isMyTurn) { // 呼び出された時の表示 statusDiv.innerHTML = `

${myNumber}番様、お呼び出し中です！すぐにお越しください。

`; } else { // 待機中の表示 statusDiv.innerHTML = `

${myNumber}番様は待機中です。あと${data.totalWaiting}組お待ちです。

`; } }
